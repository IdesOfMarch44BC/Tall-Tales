\chapter{Class Card Pools, Volume 2}

\section{The Monk \textnormal{[name pending]}}

\subsection{The Edge}
\begin{itemize}
    \item \textbf{Drawn Will} --- 1 --- L1 --- (Dup ×3) --- Blade-Strike at +1 die.
    \item \textbf{First Word} --- 1 --- L1 --- Stance --- Manifest your blade: Strikes with it use Nimbleness or Athletics dice and are lethal.
    \item \textbf{Perfect Draw} --- 1 --- L1 --- Reaction --- Manifest and Strike an enemy that just entered your reach.
    \item \textbf{The Form Holds} --- 1 --- L1 --- Stance --- +1 Defend while your blade is manifest.
    \item \textbf{Thrown Intent} --- 1 --- L1 --- Your blade strikes at 3 tiles and returns to your hand.
    \item \textbf{Cut the Material} --- 1 --- L2 --- Your blade-Strikes this round ignore 2 Armor and sever objects outright.
    \item \textbf{Two Truths} --- 1 --- L2 --- Split the will: Strike two adjacent enemies at --1 die each.
    \item \textbf{Edge Against the Volley} --- 1 --- L2 --- Reaction --- Cut one projectile or thrown attack against you out of the air.
    \item \textbf{The Long Cut} --- 1 --- L3 --- Strike every enemy in a 2-tile line once.
    \item \textbf{Cut the Working} --- 1 --- L4 --- Strike a spell, ward, or supernatural effect within reach: on 2 hits, it ends.
    \item \textbf{Cut the Footing} --- 1 --- L4 --- Strike the world instead: collapse a surface, drop a beam, open the floor beneath an enemy (save or fall/prone).
    \item \textbf{The Blade Reads Back} --- 1 --- L5 --- Cross blades deliberately: learn your opponent's training, temperament, and next intention.
    \item \textbf{The Opened Way} --- 2 --- L6 --- One-Shot --- Cut a door in distance: you and adjacent allies step up to 10 tiles, through nothing.
    \item \textbf{Cut What Binds} --- 1 --- L6 --- One-Shot --- Sever one intangible thing within reach: a possession, a compulsion, a curse, a leash. Not a debt. Not yet.
    \item \textbf{The Unnameable Edge} --- 2 --- L8 --- One-Shot --- Cut something ordinary language does not hold --- an inevitability, a connection, a certainty --- GM adjudicates; the scriptoria will want your account.
\end{itemize}

\subsection{The Eye}
\begin{itemize}
    \item \textbf{Centered Breath} --- 1 --- L1 --- (Dup ×3) --- +2 dice on your next save, or clear a --1 die penalty on yourself.
    \item \textbf{Still Water} --- 1 --- L1 --- Stance --- +2 dice on saves against fear, deception, and mind-effects; you cannot be surprised.
    \item \textbf{Read the Stance} --- 1 --- L1 --- One observed creature: learn its intent, its fear, or its next action.
    \item \textbf{Open Hand} --- 1 --- L1 --- Read a room: every present creature's mood and honesty, at a glance.
    \item \textbf{Offered Calm} --- 1 --- L1 --- An adjacent ally gains Still Water's benefits until your next surge.
    \item \textbf{Speech Without Voice} --- 1 --- L2 --- Until end of scene, your side converses silently at any distance in sight.
    \item \textbf{Unmoved} --- 1 --- L2 --- Reaction --- Automatically succeed one save against fear or mental influence.
    \item \textbf{The Noticing} --- 1 --- L2 --- One deception, disguise, or illusion within 5 tiles fails for you specifically.
    \item \textbf{Weight of Attention} --- 1 --- L3 --- Fix one creature in your regard: it takes --1 die on all actions not directed at you.
    \item \textbf{Memory Palace} --- 1 --- L3 --- Store or retrieve anything you've perceived with perfect fidelity; share one stored scene with a willing touch.
    \item \textbf{Quiet the Storm} --- 1 --- L4 --- End one rage, panic, or frenzy effect in a creature within 3 tiles (unwilling: it saves).
    \item \textbf{The Shared Stillness} --- 1 --- L4 --- Stance --- Allies within 3 tiles gain +1 die on all saves and cannot be surprised.
    \item \textbf{Going Out} --- 1 --- L5 --- One-Shot --- Leave the body for one round: perceive anywhere within the site, unseen, then return. A rehearsal, plainly labeled.
    \item \textbf{The Closed Door, Held Open} --- 2 --- L7 --- One-Shot --- With consent, enter a willing mind: heal one mental affliction, share one memory truly, or stand with them against something that got in.
    \item \textbf{The Finished Self} --- 2 --- L8 --- One-Shot --- For one scene, nothing in you is unexamined: immune to all mental influence, all your saves succeed on 3+, and you may lend one save automatically to an ally who can see your face.
\end{itemize}

\subsection{The Walker}
\begin{itemize}
    \item \textbf{Wayfarer's Step} --- 1 --- L1 --- (Dup ×3) --- Move full Move ignoring difficult ground.
    \item \textbf{Traveling Form} --- 1 --- L1 --- Stance --- +1 Move; opportunity Strikes against you lose 1 die; your blade manifests and dismisses freely.
    \item \textbf{Beggar's Bowl} --- 1 --- L1 --- The gate custom in miniature: secure food, shelter, and a hearing anywhere the tradition is known.
    \item \textbf{Correcting Touch} --- 1 --- L1 --- Adjust an ally's grip mid-action: +1 die on their current roll.
    \item \textbf{Walk On} --- 1 --- L1 --- Reaction --- When missed by an attack, move 1 tile; you were already leaving.
    \item \textbf{A Correction, Offered} --- 1 --- L2 --- An ally within 3 tiles rerolls up to 2 dice of a roll you watched.
    \item \textbf{Teaching Bout} --- 1 --- L2 --- Spar an ally a quiet hour: they gain +1 die on one skill for the session.
    \item \textbf{The Road's Judgment} --- 1 --- L2 --- Assess a stranger as walkers do: learn their training, their trouble, and whether they've sat a gate.
    \item \textbf{The Gate Custom} --- 1 --- L3 --- Once per scene per creature: an enemy accepting your offered pause stops fighting you until harmed.
    \item \textbf{Carry Word Between Houses} --- 1 --- L3 --- Once per session: know where the nearest monastery, gate-friend, or fellow walker is, and be expected there.
    \item \textbf{Both Limbs} --- 1 --- L4 --- This round, use one Edge card and one Eye card for 1 Momentum total.
    \item \textbf{Uninvited Nowhere} --- 1 --- L4 --- Stance --- You can't be barred by social station: guards hesitate, doors open, courts hear you; the robe precedes you.
    \item \textbf{Learn What We Don't Know} --- 2 --- L5 --- One-Shot --- Observe any technique used this scene: perform it once, now, at its printed cost.
    \item \textbf{The Walker's Report} --- 1 --- L6 --- One-Shot --- Once per session: the tradition acts on your word --- a house alerted, a Sundered watch begun, a Reading convened where you say.
    \item \textbf{Everywhere the Road Goes} --- 2 --- L7 --- One-Shot --- For one journey, the party moves at walker's pace: half time, no hazards you don't choose, and one useful encounter the road owed you.
\end{itemize}

\section{Tinkerer}

\subsection{The Coilwright}
\begin{itemize}
    \item \textbf{Jolt} --- 1 --- L1 --- (Dup ×3) --- 2 hits to a target within 3 tiles; constructs and Living Iron bank it instead.
    \item \textbf{Storage Cell} --- 1 --- L1 --- Consumable ×2 --- Spend for +1 Momentum (bearer's surge) or to power a device for a scene.
    \item \textbf{Patch the Line} --- 1 --- L1 --- Repair a device, Heart, or construct 2 HP; restore function to one dead machine.
    \item \textbf{Static Field} --- 1 --- L1 --- Stance --- Enemies entering tiles adjacent to you take 1 hit (vim).
    \item \textbf{Kill the Lights} --- 1 --- L1 --- Cut power in a 5-tile radius: lamps, lines, and one powered device of the GM's choice go dark.
    \item \textbf{Charged Weapon} --- 1 --- L2 --- An ally's weapon carries current: +1 hit on its next 2 hits.
    \item \textbf{Capacitor Trap} --- 1 --- L2 --- Rig a tile: first enemy entering takes 3 hits (vim).
    \item \textbf{Ground the Line} --- 1 --- L3 --- Reaction --- Redirect one vim, lightning, or Heart-driven effect within 3 tiles into the earth; bank 1 ebb.
    \item \textbf{Overvolt} --- 1 --- L3 --- A powered device or construct within 3 tiles surges: 3 hits to it, or it runs double for a round then stalls.
    \item \textbf{Jolt-Line Web} --- 1 --- L4 --- String live wire across 3 tiles: enemies crossing take 2 hits and stop.
    \item \textbf{The Quickened Heart} --- 1 --- L4 --- A vehicle, frame, or Heart you tend gains +1 Move and +1 die for the scene.
    \item \textbf{Read the Current} --- 1 --- L2 --- Trace any powered system: layout, loads, and where it fails.
    \item \textbf{Rolling Blackout} --- 2 --- L5 --- All enemy devices, constructs, and powered gear within 5 tiles lose a round.
    \item \textbf{The Dawn Clause} --- 2 --- L6 --- One-Shot --- At the procedurally correct moment (GM agrees the sky is right): your next Coilwright card triples its numbers.
    \item \textbf{Heart Transplant} --- 2 --- L7 --- One-Shot --- Field-swap a Constructive Heart: revive a dead great machine, or overdrive a live one to legendary output for a scene, then it rests a week.
\end{itemize}

\subsection{The Crucible-Trained}
\begin{itemize}
    \item \textbf{Flask of Opinion} --- 1 --- L1 --- (Dup ×3) --- Thrown, 5 tiles: 2 hits in a 1-tile burst; the ground is briefly burning, slick, or fogged (choose).
    \item \textbf{Solvent} --- 1 --- L1 --- Destroy a door's worth of material, or strip 2 Armor from a target until repaired.
    \item \textbf{Smoke Pot} --- 1 --- L1 --- 2 tiles of dense smoke: Shoot attacks through it lose 2 dice.
    \item \textbf{Litmus Sense} --- 1 --- L1 --- Identify any substance, poison, or residue by field test.
    \item \textbf{Quickmatch} --- 1 --- L1 --- Reaction --- Ignite, extinguish, or neutralize one small chemical event within 3 tiles.
    \item \textbf{Alloy on Demand} --- 1 --- L2 --- An ally's weapon gains +1 die or their Armor +1 until end of scene.
    \item \textbf{Caustic Coat} --- 1 --- L2 --- A weapon's next 3 hits also strip 1 Armor each.
    \item \textbf{Flash Powder} --- 1 --- L2 --- 1-tile burst: enemies there save or are blinded a round (--2 dice).
    \item \textbf{Transmute the Ground} --- 1 --- L3 --- 2 tiles become (choose): mire (difficult), glass (loud, slick), or brittle (collapses under weight).
    \item \textbf{The Stable Batch} --- 1 --- L3 --- Consumable ×2 --- A perfected flask: as Flask of Opinion but 3 hits, and an ally can throw it.
    \item \textbf{Neutralize} --- 1 --- L4 --- Reaction --- Cancel one acid, fire, poison, or alchemical effect as it lands, anywhere within 3 tiles.
    \item \textbf{Refiner's Eye} --- 1 --- L4 --- Assess any material's worth, purity, and provenance; spot orichalcum through a crate.
    \item \textbf{The Big Batch} --- 2 --- L5 --- 3 hits to every enemy in a 2-tile burst at 5 tiles; the crater is yours to explain.
    \item \textbf{Orichalcum Sliver} --- 2 --- L6 --- One-Shot --- Overcharge a weapon: its next attack adds hits equal to your Study.
    \item \textbf{Philosopher's Hour} --- 2 --- L7 --- One-Shot --- With a bench and an hour: transmute one material into another of like mass (stone to iron, iron to glass, lead to --- no. The Crucible has rules about that one).
\end{itemize}

\subsection{The Frame-Wright}
\begin{itemize}
    \item \textbf{The Frame} --- 1 --- L1 --- Troop --- Deploy your chassis: HP 5, Strike 3, Defend 1, Move 2, carries loads; re-musters between encounters.
    \item \textbf{Patch and Graft} --- 1 --- L1 --- (Dup ×2) --- Heal your frame (or any frame/chassis) 3 HP.
    \item \textbf{Fetch} --- 1 --- L1 --- The frame retrieves an object or drags an ally up to its Move.
    \item \textbf{Gentle the Stranger} --- 1 --- L1 --- Calm any frame, chassis, or beast of burden; +2 dice handling it thereafter.
    \item \textbf{Mounted Work} --- 1 --- L1 --- Stance --- Ride your frame: your Move becomes its Move; you gain +1 Defend from height.
    \item \textbf{Grown for This} --- 1 --- L2 --- Stance --- The frame's graft this scene: +1 Strike (horn), +1 Defend (plate), or +1 Move (limbs).
    \item \textbf{Good Beast} --- 1 --- L2 --- Reaction --- The frame interposes: an attack on you or an adjacent ally hits it instead.
    \item \textbf{Haul Away} --- 1 --- L2 --- The frame moves double its Move carrying up to two people or a wagon's load.
    \item \textbf{Trained Response} --- 1 --- L3 --- Once per surge, the frame's first action costs 0 Momentum.
    \item \textbf{Second Graft} --- 1 --- L3 --- The frame holds two Grown for This benefits at once this scene.
    \item \textbf{Stampede Line} --- 1 --- L4 --- The frame charges through enemies: each in its path takes 2 hits and is pushed aside.
    \item \textbf{Vat-Yard Triage} --- 1 --- L4 --- One-Shot --- Fully restore the frame between scenes, or return it to play mid-scene at half HP.
    \item \textbf{The Whistle They Know} --- 1 --- L5 --- Once per session: a second frame answers from somewhere plausible --- a loaner, a stray, a friend's --- for the scene.
    \item \textbf{The Yard's Best Line} --- 2 --- L6 --- One-Shot --- This scene the frame acts once per surge at 0 Momentum, and something in it is clearly, briefly, choosing to.
    \item \textbf{Grown Together} --- 2 --- L7 --- One-Shot --- For a scene, you and the frame are one working animal: share HP pools, act as one on your actions, and every drover on the frontier would pay to watch.
\end{itemize}

\section{Sawbones}

\subsection{The Red Thread}
\begin{itemize}
    \item \textbf{Suture} --- 1 --- L1 --- (Dup ×3) --- Heal an adjacent ally 2 HP.
    \item \textbf{Field Surgery} --- 1 --- L1 --- Heal an adjacent ally 3 HP and clear one bleed, poison, or injury.
    \item \textbf{Pressure Here} --- 1 --- L1 --- Reaction --- An ally within 3 tiles ignores one bleed or ongoing-damage effect this round.
    \item \textbf{Reading the Body} --- 1 --- L1 --- Diagnose anyone in sight: HP state, afflictions, cause, and one exploitable weakness.
    \item \textbf{Steady the Hands} --- 1 --- L1 --- Stance --- Your healing cards heal +1 while you weren't attacked since your last surge.
    \item \textbf{I Know Where It Hurts} --- 1 --- L2 --- Your Strike targets anatomy: on any hit, the target also loses 1 die from its next action.
    \item \textbf{Clamp and Carry} --- 1 --- L2 --- Stabilize a downed ally and move them your Move in one action.
    \item \textbf{Local Anesthetic} --- 1 --- L2 --- An ally ignores wound penalties and pain effects for a scene; the bill comes due after.
    \item \textbf{Triage} --- 1 --- L3 --- Heal 2 HP split among up to 3 allies within 3 tiles.
    \item \textbf{Case Notes} --- 1 --- L3 --- Against a creature type you've treated or dissected: your side's attacks gain +1 die vs it this scene.
    \item \textbf{Counter-Surgery} --- 1 --- L4 --- Undo flesh-work: end one graft, mutation-effect, or somatic working on a willing or held subject.
    \item \textbf{The Steady Table} --- 1 --- L4 --- Stance --- Allies adjacent to you can't be reduced below 1 HP while you have Momentum unspent on your side.
    \item \textbf{Anatomist's Strike} --- 1 --- L5 --- Strike or guide an ally's attack: it ignores Armor; you called the seam.
    \item \textbf{Not On My Table} --- 2 --- L6 --- One-Shot --- An ally at 0 HP or dead this scene is stabilized at 1 HP; the injury annex gains an entry with your name on it.
    \item \textbf{The Procedure That Gets Named} --- 2 --- L7 --- One-Shot --- With an hour and a table: repair one thing medicine says is past repair --- the ruined hand, the blinded eye, the failing organ. Documented, replicated, entered. Someone paid for this once; today it pays back.
\end{itemize}

\subsection{The Toxicologist}
\begin{itemize}
    \item \textbf{Coated Edge} --- 1 --- L1 --- (Dup ×3) --- A weapon's next 2 hits also cost the target 1 die from its next action.
    \item \textbf{Know the Antidote} --- 1 --- L1 --- Reaction --- Cancel one poison, venom, or toxin effect within reach; identify its source.
    \item \textbf{Sleep Dust} --- 1 --- L1 --- Thrown, 3 tiles: one target saves or is drowsy (--1 die, --1 Move) for 2 rounds.
    \item \textbf{Taster's Immunity} --- 1 --- L1 --- Stance --- You are immune to ingested and contact poisons; you can safely taste-test anything.
    \item \textbf{Dose the Drink} --- 1 --- L1 --- Out of combat: lace food or drink undetectably with any preparation you carry.
    \item \textbf{Weakening Agent} --- 1 --- L2 --- Touch or hit: the target's Strike and Shoot drop 1 for the scene (save resists).
    \item \textbf{Nettle Cloud} --- 1 --- L2 --- 1-tile burst at 3 tiles: enemies there take --1 die on their next roll and cough loudly.
    \item \textbf{Custom Blend} --- 1 --- L2 --- Choose your poison's rider for the scene: your Coated Edge also slows (--1 Move), sickens (--1 die saves), or marks (glows faintly).

    \item \textbf{Paralytic} --- 1 --- L3 --- Against a target already poisoned by you: it saves or loses its next action.

    \item \textbf{Blight the Blade-Line} --- 1 --- L5 --- Your side's weapons are dosed this scene: everyone's first hit each round carries your Coated Edge rider.
    \item \textbf{The Numbers Person} --- 2 --- L6 --- One-Shot --- Declare a creature type studied this session: your side's attacks vs it gain +1 die and ignore 1 Armor for the scene.
    \item \textbf{The Perfect Dose} --- 2 --- L7 --- One-Shot --- One creature, one preparation, one outcome, chosen: sleep for hours, truth for minutes, stillness for a scene, or death over days --- your call, its save, your conscience.
\end{itemize}
